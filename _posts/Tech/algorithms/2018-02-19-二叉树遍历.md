## 前序遍历
```c++
    //前序遍历
    void previsit(t *root) {

        if (root != NULL) {
            cout << root->value << endl;

            previsit(root->left);
            previsit(root->right);
        }
        else {
            //cout << "invalid" << endl;

        }
    }
```

## 中序遍历
```c++

    //中序遍历
    void midvisit(t *root) {
        if (root != NULL) {
            midvisit(root->left);
            cout << root->value << endl;
            midvisit(root->right);
        }
    }
```

## 后序遍历
```c++

    //后序遍历
    void postvisit(t *root) {
        if (root != NULL) {
            postvisit(root->left);
            postvisit(root->right);
            cout << root->value << endl;
        }
    }
```

## 层次遍历
```c++

    //层次遍历
    void levelvisit(t *root) {
        queue<t*> q;
        q.push(root);
        while (!q.empty()) {
            t *node = q.front();
            cout << node->value << endl;
            q.pop();
            if (node->left != NULL) {
                q.push(node->left);
            }
            if (node->right != NULL) {
                q.push(node->right);
            }
        }
    }
```
