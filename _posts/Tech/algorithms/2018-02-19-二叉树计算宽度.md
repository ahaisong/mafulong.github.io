## 计算最大宽度
```c++
    //计算宽度
    //可以由levercount()中改造，int book[]数组存储每层节点个数
    //也可以改造层次遍历
    int calcuwidth(t *root) {
        int lastwidth = 0, currwidth = 0;
        int nwidth = 1;
        queue<t*> q;
        lastwidth = 1;
        q.push(root);
        while (!q.empty()) {
            while (lastwidth != 0) {
                t* r = q.front();
                q.pop();
                if (r->left != NULL)
                    q.push(r->left);
                if (r->right != NULL)
                    q.push(r->right);
                lastwidth--;
            }
            currwidth = q.size();
            nwidth = nwidth > currwidth ? nwidth : currwidth;
            lastwidth = currwidth;
        }
        return nwidth;

    }
```

## 计算每一层宽度
```c++
int book[20];
void calcuWidth(node *root,int depth) {
	if (root == NULL) {
		return;
	}
	cout << root->value << "  depth: " << depth << endl;
	book[depth]++;
	if (root->left) {
		calcuWidth(root->left, depth+1);
	}
	if (root->right) {
		calcuWidth(root->right, depth+1);
	}
}
```
