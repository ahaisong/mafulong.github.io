#include <iostream>

using namespace std;

void bubbleSort(int* pData, int length)
{
	for (int i = 0; i <length; ++i)
	{
		for (int j = 0; j < length-1-i; ++j)
		{
			if (pData[i] < pData[j])
			{
				swap(pData[i], pData[j]);
			}
		}
	}
}
void print(int*, int);
//设置哨兵的bubblesort
void bubbleSort2(int *pData, int length) {
	for (int i = 0; i < length; i++) {
		bool temp = false;
		print(pData, length);
		for (int j = 0; j<length - 1 - i; j++) {
			if (pData[i] > pData[j]) {
				swap(pData[i], pData[j]);
				temp = true;
			}
		}
		if (!temp) {
			return;
		}
	}
}

void print(int* pData, int length)
{
	for (int i = 0; i != length; ++i)
	{
		cout << pData[i] << " ";
	}
	cout << endl;
}

{
