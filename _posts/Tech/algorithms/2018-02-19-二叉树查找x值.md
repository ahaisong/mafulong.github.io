## 查找x值
```c++
    //查找x值
    t* find(t* root, int x) {
        if (root == NULL) {
            return NULL;
        }
        if (root->value == x) {
            return root;
        }
        else {
            t* r = find(root->left,x);
            if (r == NULL) {
                return find(root->right, x);
            }
            else {
                return r;
            }
        }
    }
```
