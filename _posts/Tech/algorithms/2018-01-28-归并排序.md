int mergesort(vector<int> &v) {
	if (v.size() == 0) {
		for (int i = 0; i < 14; i++) {
			v.push_back(22-i);
		}
	}
	mergesort1(v);
	return 0;
}
int mergesort1(vector<int> &v) {
	if (v.size() > 1) {
		int mid = v.size() / 2;
		vector<int> v1(v.begin(), v.begin() + mid);
		vector<int> v2(v.begin() + mid, v.end());
		mergesort1(v1);
		mergesort1(v2);
		int i1 = 0, i2 = 0, i = 0;
		while (i1 < v1.size() && i2 < v2.size()) {
			if (v1[i1] < v2[i2]) {
				v[i++] = v1[i1];
				i1++;
			}
			else {
				v[i++] = v2[i2];
				i2++;
			}
		}
		if (i1 == v1.size()) {
			while (i2 < v2.size()) {
				v[i++] = v2[i2];
				i2++;
			}
		}
		else {
			while (i1 < v1.size()) {
				v[i++] = v1[i1];
				i1++;
			}
		}
	}
	return 0;
}
