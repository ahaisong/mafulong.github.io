## 插入排序
```c++
void insertSort(vector<int> &v) {
	int n = v.size();
	for (int i = 0; i < n; i++) {
		int t = v[i],j=i-1;
		for (j = i - 1; j >= 0 && v[j] > t; j--) {
			v[j + 1] = v[j];
		}
		v[j + 1] = t;
	}
}

void insertsort(vector<int> &d) {
	int n = d.size();
	for (int i = 1; i < n; i++) {
		for (int j = i; j > 0 && d[j] < d[j - 1]; j--) {
			swap(d[j], d[j - 1]);
		}
	}
}


int main()
{
	vector<int> v = { 3,2,1,4,5,8,3,5,5 };
	insertSort(v);
	for (auto it : v) {
		cout << it << " ";
	}
	return 0;
}
```
